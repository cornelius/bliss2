# bliss list — reference mockup
#
# Header fields:
#   "bliss list  Context: <name>  Path: <path>"          default view
#   "bliss list  Context: <name>  List: <list>  Path: <path>"  filtered view
#   "bliss list  Personal"                               --personal view
#   "bliss list  Personal  List: <list>"                 --personal <list>
#   "bliss list --all"                                   --all view (no numbers)
#
# Number field: right-aligned, width grows with item count.
#   1-digit (1–9):   "  N  title"   (2 spaces + digit + 2 spaces)
#   2-digit (10–99): " NN  title"   (1 space + 2 digits + 2 spaces)
#
# Section delimiters: dashes fill the number field, aligned with items.
#   1-digit zone, named:    "  ── name"
#   2-digit zone, named:    " ─── name"
#   2-digit zone, unnamed:  " ───"
#   --all (no numbers):     "  ──" / "  ── name"
#
# List order: today → this-week → next-week → later → [custom] → bugs → inbox
# Empty lists are omitted.
